const getDoctors = async (req, res) => {
  try {
    const doctors = await getAllDoctors();
    res.status(200).json({
      message: "Doctors fetched successfully",
      data: doctors
    });
  } catch (error) {
    res.status(500).json({ message: error.message });
  }
};
